### Task 2: Create `lib/linear/client.ts`

Lazy singleton `LinearClient`. Throw on missing env var (fail fast at first use, not at import time).

```typescript
import { LinearClient } from '@linear/sdk';

let _client: LinearClient | null = null;

export function getLinearClient(): LinearClient {
  if (!_client) {
    const apiKey = process.env.LINEAR_API_KEY;
    if (!apiKey) throw new Error('LINEAR_API_KEY is not set');
    _client = new LinearClient({ apiKey });
  }
  return _client;
}
```

### Task 3: Create `lib/linear/webhook-verifier.ts`

HMAC-SHA256 signature verification. Return `false` when secret is unset — never silently pass.

```typescript
import * as crypto from 'crypto';

/**
 * Verify a Linear webhook signature.
 * Returns false if secret is unset (fail-closed).
 */
export function verifyLinearWebhook(
  body: string,
  signature: string,
  secret: string
): boolean {
  if (!secret || !signature) return false;
  try {
    const hash = crypto.createHmac('sha256', secret).update(body).digest('hex');
    // Both buffers must be same length for timingSafeEqual
    const hashBuf = Buffer.from(hash, 'hex');
    const sigBuf = Buffer.from(signature, 'hex');
    if (hashBuf.length !== sigBuf.length) return false;
    return crypto.timingSafeEqual(hashBuf, sigBuf);
  } catch {
    return false;
  }
}
```
